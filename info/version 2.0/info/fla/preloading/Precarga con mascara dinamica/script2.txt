lv = new LoadVars();
lv.load("admision.swf");
// Preloader
this.onEnterFrame = function() {
	totalbytes = lv.getBytesTotal();
	bytescargados = lv.getBytesLoaded();
	totalbytes_txt.text = totalbytes;
	bytescargados_txt.text = bytescargados;
	porcentajecargado = int(bytescargados*100/totalbytes);
	porcentajecargado_txt.text = porcentajecargado+"%";
	barraprocesos._width = porcentajecargado*2;
	if (porcentajecargado == 100) {
		delete this.onEnterFrame;
		loadMovieNum("admision.swf", 0);
	}
};
